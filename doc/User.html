def first_grade?
  # 1年生以外かどうか
  return false if nil?
  this_year = Time.now.year.to_s.match(/\d{2}$/)
  current_user_year = email.match(/^\d{2}/)

  # 今の年とユーザーの学年が一致する場合はtrueを返す
  this_year[0] == current_user_year[0]
end
